  <script>
    // ---------------------------
    // Config / Elements
    // ---------------------------
    const els = {
      apiUrl: document.getElementById('apiUrl'),
      apiDot: document.getElementById('apiDot'),
      apiLabel: document.getElementById('apiLabel'),
      apiPill: document.getElementById('apiPill'),

      inputText: document.getElementById('inputText'),
      charCount: document.getElementById('charCount'),
      wordCount: document.getElementById('wordCount'),

      summarizeBtn: document.getElementById('summarizeBtn'),
      clearBtn: document.getElementById('clearBtn'),
      pasteSampleBtn: document.getElementById('pasteSampleBtn'),

      length: document.getElementById('length'),
      highlightToggle: document.getElementById('highlightToggle'),

      summaryBox: document.getElementById('summaryBox'),
      history: document.getElementById('history'),

      modeBadge: document.getElementById('modeBadge'),
      lengthBadge: document.getElementById('lengthBadge'),
      timeBadge: document.getElementById('timeBadge'),

      spinner: document.getElementById('spinner'),
      statusText: document.getElementById('statusText'),
      srStatus: document.getElementById('srStatus'),
      errorBox: document.getElementById('errorBox'),

      copyBtn: document.getElementById('copyBtn'),
      downloadBtn: document.getElementById('downloadBtn'),
    };

    let historyItems = []; // { id, mode, length, ms, summary, createdAt, snippet }
    let activeId = null;

    // ---------------------------
    // Helpers
    // ---------------------------
    function getMode(){
      return document.querySelector('input[name="mode"]:checked')?.value || 'extractive';
    }

    function setLoading(on){
      els.spinner.style.display = on ? 'inline-block' : 'none';
      els.summarizeBtn.disabled = on;
      els.clearBtn.disabled = on;
      els.pasteSampleBtn.disabled = on;
      els.apiUrl.disabled = on;
      els.length.disabled = on;
      document.querySelectorAll('input[name="mode"]').forEach(r => r.disabled = on);
    }

    function setError(msg){
      if(!msg){
        els.errorBox.style.display = 'none';
        els.errorBox.textContent = '';
        return;
      }
      els.errorBox.style.display = 'block';
      els.errorBox.textContent = msg;
      els.srStatus.textContent = msg;
    }

    function updateCounts(){
      const text = els.inputText.value || '';
      els.charCount.textContent = `${text.length.toLocaleString()} chars`;
      const words = text.trim() ? text.trim().split(/\s+/).length : 0;
      els.wordCount.textContent = `${words.toLocaleString()} words`;
    }

    function escapeHtml(str){
      return str.replace(/[&<>"']/g, s => ({
        '&':'&amp;', '<':'&lt;', '>':'&gt;', '"':'&quot;', "'":'&#039;'
      }[s]));
    }

    // Highlight extractive sentences in original textarea by rendering a "preview" in the summaryBox only,
    // since native textarea can't partially style text.
    // We'll do a lightweight sentence match: wrap matched sentences with <mark>.
    function renderOriginalWithHighlights(original, summary){
      // Only attempt highlight if user enabled it and mode extractive (where sentences likely match verbatim-ish)
      if(els.highlightToggle.value !== 'on') return null;

      const summarySentences = summary
        .split(/(?<=[.!?])\s+/)
        .map(s => s.trim())
        .filter(Boolean);

      if(summarySentences.length === 0) return null;

      // Build a safe HTML representation with marked sentences.
      // For each summary sentence, replace exact matches in the original.
      let html = escapeHtml(original);

      // Replace longest first to reduce nested replacements.
      summarySentences.sort((a,b)=> b.length - a.length).forEach(s => {
        const safe = escapeHtml(s);
        // Escape for regex
        const pattern = safe.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
        const re = new RegExp(pattern, 'g');
        html = html.replace(re, `<mark class="hl">${safe}</mark>`);
      });

      return html;
    }

    function setActiveSummary(item){
      activeId = item.id;
      els.modeBadge.textContent = `mode: ${item.mode}`;
      els.lengthBadge.textContent = `length: ${item.length}`;
      els.timeBadge.textContent = `${item.ms} ms`;

      // Display summary text
      els.summaryBox.innerHTML = `<div>${escapeHtml(item.summary)}</div>`;

      // Enable actions
      els.copyBtn.disabled = false;
      els.downloadBtn.disabled = false;

      // Update active highlight in history list
      renderHistory();

      // If extractive highlight enabled, show a hint below summary by injecting a collapsible preview of highlighted original
      // (keeps dedicated summary panel as the main display; preview is optional for comparison).
      const original = els.inputText.value || '';
      if(item.mode === 'extractive'){
        const highlighted = renderOriginalWithHighlights(original, item.summary);
        if(highlighted){
          const details = document.createElement('details');
          details.style.marginTop = '10px';
          details.innerHTML = `
            <summary class="small" style="cursor:pointer;">Show original text with extractive highlights</summary>
            <div style="margin-top:8px; padding:10px; border:1px solid var(--line); border-radius:12px; background:rgba(0,0,0,0.06); max-height:220px; overflow:auto;">
              <div style="white-space:pre-wrap; line-height:1.55;">${highlighted}</div>
            </div>
          `;
          els.summaryBox.appendChild(details);
        }
      }
    }

    function renderHistory(){
      els.history.innerHTML = '';
      if(historyItems.length === 0){
        els.history.innerHTML = `<div class="small" style="padding: 10px 12px;">No summaries yet. Generate one to build a comparison history.</div>`;
        return;
      }

      historyItems.forEach(item => {
        const row = document.createElement('div');
        row.className = 'historyItem' + (item.id === activeId ? ' active' : '');
        row.tabIndex = 0;
        row.setAttribute('role', 'button');
        row.setAttribute('aria-label', `Summary history item ${item.mode} ${item.length}`);

        row.innerHTML = `
          <div>
            <div class="historyTitle">
              <strong>${item.mode}</strong>
              <span class="badge">len: ${item.length}</span>
              <span class="badge">${item.ms} ms</span>
              <span class="badge">${item.createdAt}</span>
            </div>
            <div class="historySnippet">${escapeHtml(item.snippet)}</div>
          </div>
          <div class="historyRight">Click to view</div>
        `;

        row.addEventListener('click', () => setActiveSummary(item));
        row.addEventListener('keydown', (e) => {
          if(e.key === 'Enter' || e.key === ' ') { e.preventDefault(); setActiveSummary(item); }
        });

        els.history.appendChild(row);
      });
    }

    function downloadText(filename, text){
      const blob = new Blob([text], { type: 'text/plain;charset=utf-8' });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = filename;
      document.body.appendChild(a);
      a.click();
      a.remove();
      URL.revokeObjectURL(url);
    }

    // ---------------------------
    // API calls
    // ---------------------------
    async function checkApi(){
      // Lightweight check: HEAD/GET to endpoint may not be supported.
      // We'll do a GET with ?ping=1; if it fails, we mark offline but UI still works.
      const url = els.apiUrl.value.trim();
      if(!url) return;

      try{
        const res = await fetch(url + (url.includes('?') ? '&' : '?') + 'ping=1', { method: 'GET' });
        // Any response is considered "reachable"
        els.apiDot.classList.remove('offline');
        els.apiLabel.textContent = 'API: reachable';
        els.apiPill.title = 'Backend endpoint responded to ping.';
      }catch{
        els.apiDot.classList.add('offline');
        els.apiLabel.textContent = 'API: not reachable (UI still OK)';
        els.apiPill.title = 'Could not reach backend endpoint. Summarization requests will fail until backend is running.';
      }
    }

    async function generateSummary(){
      const text = (els.inputText.value || '').trim();
      if(!text){
        setError('Please paste or type some text to summarize.');
        return;
      }

      setError(null);
      setLoading(true);
      els.statusText.textContent = 'Generating summary…';
      els.srStatus.textContent = 'Generating summary';

      const mode = getMode();
      const length = els.length.value;
      const url = els.apiUrl.value.trim();

      const t0 = performance.now();
      try{
        const res = await fetch(url, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ text, mode, length })
        });

        if(!res.ok){
          const errText = await res.text().catch(()=> '');
          throw new Error(`HTTP ${res.status} ${res.statusText}${errText ? ' — ' + errText : ''}`);
        }

        const data = await res.json();
        // Expect: { summary: "..." } (or { result: "..."}). We'll accept either.
        const summary = (data.summary ?? data.result ?? '').toString();

        const ms = Math.round(performance.now() - t0);
        if(!summary){
          throw new Error('Backend returned an empty summary. Ensure your API returns JSON like { "summary": "..." }.');
        }

        const item = {
          id: crypto.randomUUID(),
          mode,
          length,
          ms,
          summary,
          createdAt: new Date().toLocaleTimeString([], {hour:'2-digit', minute:'2-digit'}),
          snippet: summary.length > 140 ? summary.slice(0, 140) + '…' : summary
        };

        historyItems = [item, ...historyItems].slice(0, 25); // keep last 25
        renderHistory();
        setActiveSummary(item);

        els.statusText.textContent = 'Done.';
        els.srStatus.textContent = 'Summary generated';
      }catch(err){
        setError(`Could not generate summary: ${err.message}`);
        els.statusText.textContent = 'Error.';
      }finally{
        setLoading(false);
      }
    }

    // ---------------------------
    // UI actions
    // ---------------------------
    els.inputText.addEventListener('input', updateCounts);
    els.apiUrl.addEventListener('change', checkApi);

    els.summarizeBtn.addEventListener('click', generateSummary);

    els.clearBtn.addEventListener('click', () => {
      els.inputText.value = '';
      updateCounts();
      els.summaryBox.innerHTML = `<span class="small">Your generated summary will appear here.</span>`;
      els.modeBadge.textContent = 'mode: —';
      els.lengthBadge.textContent = 'length: —';
      els.timeBadge.textContent = '— ms';
      historyItems = [];
      activeId = null;
      renderHistory();
      els.copyBtn.disabled = true;
      els.downloadBtn.disabled = true;
      setError(null);
      els.statusText.textContent = 'Cleared.';
    });

    els.pasteSampleBtn.addEventListener('click', () => {
      els.inputText.value =
`Natural language processing (NLP) is a field of artificial intelligence focused on enabling computers to understand and generate human language. Modern NLP systems combine statistical methods with deep learning models to perform tasks such as summarization, translation, and question answering. Extractive summarization selects key sentences from a document, while abstractive summarization generates new sentences that paraphrase the original content. Controlling summary length helps users quickly choose between brief overviews and more detailed digests. In practice, applications often provide multiple modes and lengths to support different reading goals, from skimming to deep review.`;
      updateCounts();
      setError(null);
      els.statusText.textContent = 'Sample text pasted.';
    });

    els.copyBtn.addEventListener('click', async () => {
      const item = historyItems.find(h => h.id === activeId);
      if(!item) return;
      try{
        await navigator.clipboard.writeText(item.summary);
        els.statusText.textContent = 'Copied summary to clipboard.';
        els.srStatus.textContent = 'Copied';
      }catch{
        setError('Copy failed (clipboard permissions). You can select text and copy manually.');
      }
    });

    els.downloadBtn.addEventListener('click', () => {
      const item = historyItems.find(h => h.id === activeId);
      if(!item) return;
      const name = `summary_${item.mode}_${item.length}.txt`;
      downloadText(name, item.summary);
      els.statusText.textContent = `Downloaded ${name}.`;
    });

    // Init
    updateCounts();
    renderHistory();
    checkApi();
  </script>
