    <h1>Smart Document Summarizer</h1>
    <p>Upload PDF, DOCX, or TXT files to begin processing.</p>

    <!-- Clickable upload area linked to hidden file input -->
    <label class="upload-box" for="fileInput">
        <strong>Click to Upload</strong><br>
        or drag & drop your file here
    </label>

    <!-- Hidden file input (triggered by clicking the label) -->
    <input type="file" id="fileInput" accept=".pdf,.docx,.txt">

    <!-- Status display area (hidden initially) -->
    <div id="status">
        <div class="spinner"></div>
        <span id="statusText">Processing your file...</span>
    </div>
</div>

<script>
    // Get references to HTML elements
    const fileInput = document.getElementById("fileInput");
    const statusBox = document.getElementById("status");
    const spinner = document.querySelector(".spinner");
    const statusText = document.getElementById("statusText");

    // Function to change background color dynamically
    function setBackground(color) {
        document.body.style.backgroundColor = color;
    }

    // Set initial background color on page load
    setBackground("#f5f7fa");

    // Event listener for when a file is selected
    fileInput.addEventListener("change", () => {

        // Check if a file was selected
        if (fileInput.files.length > 0) {

            // Get the first selected file
            const file = fileInput.files[0];

            // Change background to indicate processing
            setBackground("#fff4d6");

            // Show status box and spinner
            statusBox.style.display = "block";
            spinner.style.display = "block";

            // Display file name in status text
            statusText.textContent = `Processing "${file.name}"...`;

            // Simulate processing delay (2 seconds)
            setTimeout(() => {

                // Hide spinner after "processing"
                spinner.style.display = "none";

                // Update status message
                statusText.textContent = "File uploaded successfully! Ready for extraction.";

                // Change background to green (success state)
                setBackground("#e6ffe6");

            }, 2000);
        }
    });
</script>
